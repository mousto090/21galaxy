Project structure
.
├── bin                     # bin
│   ├── www                 # App starting point
|
├── public                  # public files
│   ├── dist                # Assests generated by Webpack 
│       ├── css             # CSS generated by Webpack (do not modify)
│       ├── fonts           # Fonts generated by Webpack (do not modify)
│       ├── images          # all images goes here generated or not
│       ├── js              # JS files generated by Webpack (do not modify)
|
├── routes                  # App routes
│   └── routes.js           # Routers
│   └── ...                 # Other routes

├── src                     # Real app codes (all js and styles must be done here)
│   ├                       # Webpack generate from here to public/dist
│   ├── fonts               # Fonts
│   ├── images              # Images (to use only in .html or .scss files)
│   └── js                  # JS
│   └── scss                # App styles 
├── views                   # All ejs views goes here
|
├── pm2.js                  # pm2 init(TO DO: add pm2 config, in prod pm2 is used without config file)
├── webpack.config.js       # pm2 configuration file
├── package.json           
├── README.md         
└── app.js                  # App configuration
